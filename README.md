```text
DJANGO_SETTING_POSTGRES_DB=db_name_of_your_db
DJANGO_SETTING_POSTGRES_USER=user_name_of_your_db

DJANGO_SETTING_HOST=host_of_your_db
DJANGO_SETTING_PORT=port_of_your_db
```
